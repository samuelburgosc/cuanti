> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Agregar un sistema de clientes donde cada venta puede tener un cliente asignado (opcional), con una tab "Clientes" en Historial que muestra la lista con total gastado e historial por cliente.

**Architecture:** Todo en `index.html`. Nueva tabla `clientes` en Supabase + FK `cliente_id` en `ventas`. Cache `clientesDB` cargado al inicio. Tab "Clientes" dentro de la pantalla Historial existente. Toggle opcional "¿Quién compra?" en Paso 2 de venta con autocomplete sobre `clientesDB`.

**Tech Stack:** HTML/CSS/JS vanilla, Supabase (PostgreSQL). Verificación manual en navegador.

```javascript
function renderClientes() {
  const lista = document.getElementById('clientes-lista');
  if (!lista) return;
  const q = (document.getElementById('clientes-search-input')?.value || '').toLowerCase();

  // Calcular stats de cada cliente desde ventasDB
  const stats = {};
  ventasDB.filter(v => v.cliente_id).forEach(v => {
    if (!stats[v.cliente_id]) stats[v.cliente_id] = { n: 0, total: 0 };
    stats[v.cliente_id].n++;
    stats[v.cliente_id].total += v.gan;
  });

  let clientes = clientesDB.filter(c =>
    !q ||
    c.nombre.toLowerCase().includes(q) ||
    (c.telefono || '').includes(q)
  );

  if (clientes.length === 0) {
    lista.innerHTML = `
      <div style="text-align:center;padding:40px 20px;">
        <div style="font-size:36px;margin-bottom:12px;">👥</div>
        <div style="font-size:14px;font-weight:600;margin-bottom:6px;">Sin clientes registrados</div>
        <div style="font-size:12px;color:var(--muted);">Asígnalos al hacer una venta.</div>
      </div>`;
    return;
  }

  // Ordenar por total gastado descendente
  clientes = clientes.slice().sort((a, b) =>
    (stats[b.id]?.total || 0) - (stats[a.id]?.total || 0)
  );

  lista.innerHTML = clientes.map(c => {
    const st = stats[c.id] || { n: 0, total: 0 };
    return `
    <div onclick="renderDetalleCliente(${c.id})"
      style="background:var(--sf);border:1px solid var(--bd);border-radius:12px;padding:14px 16px;margin-bottom:8px;cursor:pointer;display:flex;justify-content:space-between;align-items:center;">
      <div style="flex:1;min-width:0;">
        <div style="font-size:14px;font-weight:600;margin-bottom:3px;">${c.nombre}</div>
        <div style="font-size:11px;color:var(--muted);">${c.telefono ? c.telefono + ' · ' : ''}${st.n} compra${st.n !== 1 ? 's' : ''}</div>
      </div>
      <div style="text-align:right;flex-shrink:0;">
        <div style="font-family:'DM Mono',monospace;font-size:14px;font-weight:700;color:var(--green);">$${fmt(st.total)}</div>
        <div style="font-size:10px;color:var(--muted);">total ganado</div>
      </div>
    </div>`;
  }).join('');
}
```

```javascript
function ventaToggleCliente() {
  const panel  = document.getElementById('v-cliente-panel');
  const icon   = document.getElementById('v-cliente-toggle-icon');
  const toggle = document.getElementById('v-cliente-toggle');
  const abierto = panel.style.display !== 'none';
  panel.style.display = abierto ? 'none' : '';
  icon.textContent    = abierto ? '＋' : '−';
  toggle.style.borderColor = abierto ? 'var(--bd2)' : 'var(--accent)';
  if (abierto) ventaLimpiarCliente();
}

function ventaBuscarCliente(q) {
  ventaState.clienteId     = null;
  ventaState.clienteNombre = q.trim() || null;
  const sugs = document.getElementById('v-cliente-sugerencias');
  if (!sugs) return;
  if (q.length < 2) { sugs.style.display = 'none'; return; }

  const matches = clientesDB.filter(c =>
    c.nombre.toLowerCase().includes(q.toLowerCase()) ||
    (c.telefono || '').includes(q)
  ).slice(0, 5);

  if (matches.length === 0) { sugs.style.display = 'none'; return; }

  sugs.style.display = '';
  sugs.innerHTML = matches.map(c => {
    const nombreSafe = c.nombre.replace(/'/g, '&#39;');
    return `<div onclick="ventaSelCliente(${c.id}, '${nombreSafe}')"
      style="padding:10px 12px;border-bottom:1px solid var(--bd);cursor:pointer;font-size:13px;">
      <span style="font-weight:600;">${c.nombre}</span>
      ${c.telefono ? `<span style="font-size:11px;color:var(--muted);margin-left:6px;">${c.telefono}</span>` : ''}
    </div>`;
  }).join('');
}

function ventaSelCliente(id, nombre) {
  ventaState.clienteId     = id;
  ventaState.clienteNombre = nombre;
  const input = document.getElementById('v-cliente-input');
  const sugs  = document.getElementById('v-cliente-sugerencias');
  const sel   = document.getElementById('v-cliente-seleccionado');
  const selNombre = document.getElementById('v-cliente-sel-nombre');
  if (input) input.value = '';
  if (sugs)  sugs.style.display = 'none';
  if (sel)   sel.style.display  = '';
  if (selNombre) selNombre.textContent = nombre;
}

function ventaLimpiarCliente() {
  ventaState.clienteId     = null;
  ventaState.clienteNombre = null;
  const input = document.getElementById('v-cliente-input');
  const sugs  = document.getElementById('v-cliente-sugerencias');

  if (input) input.value = '';
  if (sugs)  sugs.style.display = 'none';
```
